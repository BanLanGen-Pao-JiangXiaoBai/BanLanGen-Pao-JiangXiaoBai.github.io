# 简易的Eventbus(真正的EventBus应该还会有this绑定)
    var eventbus = (function(){
        var types={

        }
        return {
            $on:function(type,fn){
                if(types[type]) {
                    types[type].push(fn)
                }else {
                    types[type]=[fn]
                }
            },
            $emit:function(type) {
                types[type].forEach(item => {
                    item()
                });
            }
        }
    })()
    eventbus.$on('demo',function(){
        console.log('demo')
    })
    eventbus.$on('demo',function(){
        console.log('demo1')
    })
    eventbus.$emit('demo')

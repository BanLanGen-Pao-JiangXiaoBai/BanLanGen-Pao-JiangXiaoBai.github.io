# 监听父元素宽高插件（onresize方法只有body才有）

# element-resize-detector插件可以监听父元素的宽高变化执行回调函数，底层是通过ResizeObserver、与MutationObserver实现的，ResizeObserver只要内容区域发生了变化就可以监听到，MutationObserver监听dom元素的属性变化，如果宽高是用%定义的，实际的内容区域大小改变是监听不到的，只能监听到dom元素的值发生变化，属性的改变或者添加删除了子元素。

# element-resize-detector这个插件有个坑就是一个dom元素只能注册一个监听函数，类似单例，如果一个地方监听了某个dom元素并添加的监听事件A，另一个地方也要监听这个dom元素并添加事件B，当这个元素宽高变化只会执行事件A，要想实现一个dom元素挂载多个监听事件需要进行处理，把需要执行的方法都放到某个数组中保存，执行的时候循环取出数组中的方法并调用。

const addResize = function(el, fn) {
  if (!el.__resizeListeners__) {
    // 被监听的元素下定义一个存储方法的数组
    el.__resizeListeners__ = []
    el.__ro__ = elementResizeDetectorMaker()
    el.__ro__.listenTo(el, function() {
        // 循环方法数组执行方法
      if (el.__resizeListeners__.length) {
        el.__resizeListeners__.forEach(fn => {
          fn()
        })
      }
    })
  }
  el.__resizeListeners__.push(fn)
}

const remove = function(el, fn) {
  if (!el || !el.__resizeListeners__) return
  el.__resizeListeners__.splice(el.__resizeListeners__.indexOf(fn), 1)
  if (!el.__resizeListeners__.length) {
    el.__ro__.disconnect()
  }
}

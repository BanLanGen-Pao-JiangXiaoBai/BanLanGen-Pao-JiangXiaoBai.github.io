    # node-xlsx模块与json2xls模块


    利用这俩个模块将list.xlsx文件中的数据以性别分类并生成俩个新的分好类的xlsx文件
    list.xlsx文件 

    const fs = require('fs')
const json2xls = require('json2xls')
const xlsx = require('node-xlsx');
var obj = xlsx.parse('list.xlsx')
console.log(obj[0].data[0])
//obj是读取到list.xlsx的数据，obj[0]表示list.xlsx中的第一个sheet文档
//obj[0].data[0]表示文档中的第一行数据，obj[0].data[0][2]表示第一行数据的第三列
var man=[]
var women=[]
for(var i=0;i<obj[0].data.length;i++){
    if(obj[0].data[i][2]=='男性'){
        man.push(obj[0].data[i])
    }else{
        women.push(obj[0].data[i])
    }
}
let manxls=json2xls(man)//将二维数组作为参数传入json2xls，再掉用writeFlie生成文件
let womenxls=json2xls(women)
fs.writeFileSync('man.xlsx',manxls,"binary")
fs.writeFileSync('women.xlsx',womenxls,"binary")

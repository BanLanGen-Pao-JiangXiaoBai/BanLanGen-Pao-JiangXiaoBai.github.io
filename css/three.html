    # position

            *{
            margin: 0;
            padding: 0;
        }
        .left{
            width: 300px;
            background-color: blueviolet;
            position: absolute;
            top: 0;
            left: 0px;
        }
        /* 水平或者垂直方向的偏移属性缺少，水平方向或垂直方向会和原来文档流的位置一样 */
        .right{
            position: absolute;
            top: 0;
            right: 0px;
            width: 300px;
            background-color: burlywood;
        }
        .main{
            background-color: chocolate;
            margin: 0 300px;
        }
        div{
            height: 100px;
        }

        <div class="left"></div>
    <div class="main">position</div>
    <div class="right"></div>

    # flex

            .box{
            display: flex;
            /* width: 100%; */
        }
        .left{
            /* width: 300px; */
            flex: 0 1 300px;
            background-color: blueviolet;
        }
        .right{
            flex: 0 1 300px;
            /* width: 300px; */
            background-color: burlywood;
        }
        .main{
            flex: auto;
            background-color: chocolate;
        }
        div{
            height: 100px;
        }

        <div class="box">
        <div class="left"></div>
        <div class="main">11</div>
        <div class="right"></div>
    </div>

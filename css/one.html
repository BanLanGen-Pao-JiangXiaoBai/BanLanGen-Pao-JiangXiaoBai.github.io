/*循环读取map的函数生成box-shadow*/
@function xiangsu($a,$size){
	$sh:'';
	$rows:length($a);
	@for $row from 1 through $rows{
		$row-num:nth($a,$row);
		@for $col from 1 through length($row-num){
			$dot:nth($row-num,$col);
			$sh:$sh + $size*$col + ' ' + $size*$row + ' ' + map-get($color,$dot);
			@if not (($col==length($row-num)) and ($row==$rows)){
				$sh:$sh+',';
			} 
		}
	}
	@return unquote($sh)
};

    box-shadow总结： box-shadow默认的大小是原图形的大小，box-shadow的第四个参数每增加1px,阴影的大小向四个方向个加1px,scss中的$size一定要与原图形的长宽相同，代表的是偏移步长，如果不同，会因为层叠效果而变形，后绘制的与先绘制的层叠，优先显示先绘制的，第一行绘制的会因为他前面没有层叠而变形，格子像素拉长，而其余行因为有前面的层叠不会变形还是像素格子大小。

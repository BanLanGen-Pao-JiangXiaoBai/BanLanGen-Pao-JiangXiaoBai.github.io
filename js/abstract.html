# js中的抽象类是通过原型链实现的，不会在编译的时候报错，而是在调用的时候才报错。通过让子类继承了抽象类(把抽象类设置到子类的原型链上),当子类没有实现抽象类的方法的时候，子类调用的便是抽象类中的方法，抽象类的方法throw Error()

    function Car(){
        //Car抽象类
    }
    Car.prototype.getPrice=function(){
        throw new Error('抽象方法不能调用')
    }
    Car.prototype.setPrice=function(){
        throw new Error('抽象方法不能调用')
    }
    // Car的实现类
    function CarInstance(price){
        this.price=price
    }
    CarInstance.prototype=new Car()
    // 修改了prototype对象 需要重新定义constructor
    CarInstance.prototype.constructor=CarInstance
    // 抽象方法的实现
    CarInstance.prototype.getPrice=function(){
        return this.price
    }
    CarInstance.prototype.setPrice=function(price){
        this.price=price
    }
    var car=new CarInstance(11111)
    console.log('修改前price',car.getPrice()) // 11111     如果抽象方法没有实现会报错:  Error:抽象方法不能调用
    car.setPrice(22222)
    console.log('修改后price',car.getPrice()) // 22222

# 抽象类进行工厂化 将抽象类添加到工厂中，将抽象类绑定到传入的子类的原型上

    function Factory(instance/*类对象*/,abstructor/*String*/){
        // 将abstructor添加到instance的原型链上
        function extend(){
            function F(){
                // 过渡类
            }
            F.prototype=new Factory[abstructor]()
            instance.prototype=new F()
            // 重新定义了prototype需要手动添加constructor
            instance.prototype.constructor=instance
        }
        typeof Factory[abstructor]==='function'?extend():(function(){throw new Error('没有该抽象类')}())
    }
    // 为Factory添加抽象类
    Factory.Car=function(){
        this.type='car'
    }
    Factory.Car.prototype.getPrice=function(){
        throw new Error('抽象方法不能调用')
    }
    Factory.Car.prototype.setPrice=function(){
        throw new Error('抽象方法不能调用')
    }
    //Bus抽象类
    Factory.Bus=function(){
        this.type='car'
    }
    Factory.Bus.prototype.getPrice=function(){
        throw new Error('抽象方法不能调用')
    }
    Factory.Bus.prototype.setPrice=function(){
        throw new Error('抽象方法不能调用')
    }
    // 抽象类子类
    function BMW(price){
        this.price=price
    }
    // 将抽象类添加到子类原型上
    Factory(BMW,'Car')
    // 子类实现抽象方法
    BMW.prototype.getPrice=function(){
        return this.price||'未设置'
    }
    BMW.prototype.setPrice=function(price){
        this.price=price
    }
    var bmw=new BMW()
    console.log(bmw.getPrice())//未设置
    bmw.setPrice(22222)
    console.log(bmw.getPrice())//22222

# 模块化编程
# 今天改bug遇上个在模块内定义了变量，在export导出产生类似闭包与单例的情况，重复创建多个组件，多个组件都用的是一个变量，改变一个影响其他的变量。
# 用es6的import/export试了一下
// es_a.js
var num = 0
function add() {
    num++
}
function get() {
    // console.log('get方法', num)
    return num
}
export { num, add, get }

// es_b.js b中引入a模块导出的变量与函数
import { num, add, get } from './es_a.js'
console.log('b文件')
console.log('b文件中输出的num的值', num) // 输出a中自己导出的num 0
add()
console.log('b文件中输出的num的值', num) // 输出a中自己导出的num 1
console.log('b文件中输出的get num的值', get()) // 输出a中方法获得的num 1

// es_c.js c中引入a模块  并引入b模块使b模块执行改变num值观察C中的变化
import { num, add, get } from './es_a.js'
import './es_b'
console.log('c文件')
console.log('c文件中输出的num的值', num) // 输出a中自己导出的num 1
add() // 将num + 1
console.log('c文件中输出的num的值', num) // 输出a中自己导出的num 2
console.log('c文件中输出的get num的值', get()) // 输出a中方法获得的num 2

# 这也是vue中data是用方法return返回对象的原因 防止单例 修改一个影响其他
# 用node exports/require
// amd_a.js
var num = 0
function add() {
    num++
}
function get() {
    // console.log('get方法', num)
    return num
}
module.exports = { num, add, get }

//amd_b.js
var a = require('./amd_a')
console.log('b文件')
console.log('b文件中输出的num的值', a.num) // 0
a.add()
console.log('b文件中输出的num的值', a.num) // 0
console.log('b文件中输出的get num的值', a.get()) // 1
module.exports.a = a

// amd_c.js
var a=require('./amd_a')
var b =require('./amd_b')
console.log('C文件')
console.log('C文件', a.num) // 0
a.add()
console.log('C文件', a.num) // 0
console.log('C文件 get num', a.get()) // 2

# node exports/require 的输出中并不会改变导出的num,导出的num是被拷贝的,而方法对原文件中num的修改会影响到别的文件

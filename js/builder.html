# 建造者模式是化整为零，将一个大的对象分为几个小的模块，就像整个项目分为各个功能点实现，最后再合成最终对象。

            // 配置一台电脑，需要显示屏，主机，鼠标，键盘
        function Computer(type){
            this.type=type||'组装机'
        }
        Computer.prototype.getType=function(){
            return this.type
        }
        // 屏幕
        function Screen(type){
            this.type=type
        }
        Screen.prototype.setType=function(type){
            this.type=type
        }
        // 鼠标
        function Mouse(type){
            this.type=type
        }
        Mouse.prototype.setType=function(type){
            this.type=type
        }
        // 键盘
        function Keyboard(type){
            this.type=type
        }
        Keyboard.prototype.setType=function(type){
            this.type=type
        }
        //主机
        function Host(type){
            this.type=type
        }
        Host.prototype.setType=function(type){
            this.type=type
        }
        // 组合电脑
        function NewComputer({ComputerType/* 可选参数*/,ScreenType,MouseType,KeyboardType,HostType}={}){
            var computer
            if(ComputerType){
                computer=new Computer(ComputerType)
                computer.screen=new Screen(ComputerType)
                computer.mouse=new Mouse(ComputerType)
                computer.keyboard=new Keyboard(ComputerType)
                computer.host=new Host(ComputerType)
                return computer
            }else{
                computer=new Computer()
                computer.screen=new Screen(ScreenType)
                computer.mouse=new Mouse(MouseType)
                computer.keyboard=new Keyboard(KeyboardType)
                computer.host=new Host(HostType)
                return computer
            }
        }
        var computer=NewComputer({ComputerType:'联想'})

# 暂时没有很好的例子，把Computer分为几个子模块进行组合，子模块之间的联系也比较少，一个模块代码需要修改不需要考虑影响其他模块，最近改代码深有体会，耦合高的代码改起来真是让人头疼。

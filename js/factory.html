    # 安全工厂模式

    # 安全是指类方法不论是通过new创建实例还是直接调用都能正确的创建出实例

            function Demo(x,y){
            this.x=x
            this.y=y
        }
        var demo=Demo(1,2) //demo :undefined
        var demo1=new Demo(1,2)//demo1: {x:1,y:2}

    # 这个例子中，如果不小心这样创建了demo，this指向window会污染window,而demo1是正确的方式，想要用调用方式创建出Demo类的实例就必须做一些判断，instanceof是通过原型链判断是否是某个类的实例

            function Demo(x,y){
            if(this instanceof Demo){
                this.x=x
                this.y=y
                return
            }
            return new Demo(x,y)
        }
        var demo=Demo(1,2)//demo:{x:1,y:2}
        var demo1=new Demo(1,2)//demo1:{x:1,y:2}

    # 通过判断可以用调用方式正确的创建出实例了

    # 工厂则是指生产专门处理某一类功能的对象，创建某一类具有比较多相似之处的类的实例。调用者不需要知道其内部的基类,只要关心这个Factory函数就能够实例化想要的对象。

            var Factory=(function(){
            var yifu=function(size){
                this.size=size
            }
            yifu.prototype.getType=function(){
                return '衣服'
            }
            var kuzi=function(size){
                this.size=size
            }
            kuzi.prototype.getType=function(){
                return '裤子'
            }
            return function(type,size){
                switch(type){
                    case 'yifu':
                        return new yifu(size)
                    case 'kuzi':
                        return new kuzi(size)
                }
            }
        }()) 
        var yifu=Factory('yifu','175')
        var kuzi=Factory('kuzi','185')
        console.log(yifu.getType())//衣服
        console.log(kuzi.getType())//裤子
        console.log(yifu)//{size:'175'}
        console.log(kuzi)//{size:'185'}

    # 匿名函数自调用返回工厂函数，利用闭包将基类隐藏起来避免全局污染,基类是通过程序new的,避免了有可能的当函数直接调用。然而这种方式的工厂如果要添加新的基类，需要在匿名函数内创建基类还要在返回函数中再进行创建实例，有违背'懒人推动社会进步原则'(所有的便利都是为了懒人少做功而产生的),所以还需要更简化一点。

    # 可以将返回函数变成动态创建某个类实现,而传入的type仅仅是一个字符串，不能直接拿来new，需要拿到指向基类的索引进行new,而基类在闭包中实现的，外部拿不到，所以要么将基类创建在外面，但是会多创建几个变量，最好的方法是直接创建在工厂函数Factory的prototype中,或者直接创建在Factory函数对象下，Factory.xxx=function(){}

            // 工厂函数采用安全模式，无论new和直接调用方法都可以进行创建实例
        var Factory=function(type,size){
            if(this instanceof Factory){
                return new this[type](size)
            }
            return new Factory(type,size)
        }
        Factory.prototype.yifu=function(size){
                this.size=size
        }
        //为衣服类添加公有方法
        Factory.prototype.yifu.prototype.getType=function(){
                return '衣服'
        }
        Factory.prototype.kuzi=function(size){
                this.size=size
        }
        // 为裤子类添加公有方法
        Factory.prototype.kuzi.prototype.getType=function(){
                return '裤子'
        }
        //new 方式创建实例
        var yifu=new Factory('yifu','175')
        var kuzi=new Factory('kuzi','185')
        // 函数调用方式创建实例
        var yifu1=Factory('yifu','175')
        console.log(yifu.getType())//衣服
        console.log(kuzi.getType())//裤子
        console.log(yifu)//{size:'175'}
        console.log(yifu1)//{size:'175'}
        console.log(kuzi)//{size:'185'}
